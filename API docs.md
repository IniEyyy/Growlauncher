## NetAvatar

```lua
local NetAvatar = {
    posX,
    posY,
    netID,
    name,
    userID,
    country,
    punchID,
    guildflag,
    isLeft,
    posXenc,
    posYenc,
    sizeX,
    sizeY,
    sizeXenc,
    sizeYenc,
    waterSpeed,
    status,
    irisColor --[[= {vec4}]],
    pupilColor --[[= {vec4}]],
    equip = {
        hair,
        shirt,
        pants,
        feet,
        hand,
        back,
        face,
        mask,
        necklace
    },
    effect = {
        hair,
        shirt,
        pants,
        feet,
        hand,
        back,
        face,
        mask,
        necklace
    }
}
```
